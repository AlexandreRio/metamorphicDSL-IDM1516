 - $java -jar IDM_homework_Championjar (chemin vers fichier pgn)*
	(java -jar IDM_homework_Championjar ../db-kas96.pgn)
 - $python pgnChampion.py
	(pour changer les chemins edité les 2 derniéres lignes de pgnChampion.py)
